import {afficherToast, corriger} from '/composant/fonction/afficher.js';

const btnModifier = document.getElementById('btnModifier');
const email = document.getElementById('email');

btnModifier.onclick = () => {
    if (email.checkValidity()) {
        // sauvegarde de l'email avant modification
        email.dataset.old = email.value;
        afficherToast("Modification prise en compte", "info");
    } else {
        // restauration de l'email
        corriger(email, email.validationMessage , email.dataset.old);
    }
}

// alimentation de l'email avec une valeur par dÃ©faut
email.value = "guy.verghote@saint-remi.net";
email.dataset.old = email.value;
