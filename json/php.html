Le script php.html affiche l'entête du tableau
<table class='table'>
      <thead>
         <tr>
              <th class='col-5'>Nom</th>
              <th class='col-5'>Prénom</th>
              <th class='col-2 text-center'>Option</th>
        </tr>
     </thead>
     <tbody id="lesLignes"></tbody>

 Le script php.js génère les balises tr et td pour les intégrer sur l'interface
 for (const etudiant of lesEtudiants) {
       const tr = lesLignes.insertRow();
       tr.insertCell().innerText = etudiant.nom;
       tr.insertCell().innerText = etudiant.prenom;
       const td = document.createElement('td');
       td.classList.add('text-center');
       // td.style.textAlign = 'center';
       td.innerText = etudiant.option;
       tr.appendChild(td);
}
